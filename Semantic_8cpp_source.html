258void SemanticAnalyzer::CheckFunctionCall(const std::string& funcName) {
259 if (functionSignatures_.find(funcName) == functionSignatures_.end()) {
260 throw std::runtime_error("Undefined function: " + funcName);
261 }
262 if (curLex_.get_text() != "(") {
263 throw std::runtime_error(
264 "Expected '(' after function name: " + funcName +
265 "\nOn line: " + std::to_string(curLex_.get_line()));
266 }
267 GetLexem();
268 std::vector<std::string> argTypes;
269 while (curLex_.get_text() != ")") {
270 if (curLex_.get_type() == "NUMBER") {
271 argTypes.push_back((curLex_.get_text().find('.') != std::string::npos)
272 ? "float"
273 : "int");
274 } else if (curLex_.get_type() == "STRING") {
275 argTypes.push_back("string");
276 } else if (curLex_.get_type() == "IDENTIFIER") {
277 if (!IsVarDefined(curLex_.get_text())) {
278 throw std::runtime_error(
279 "Undefined variable: " + curLex_.get_text() +
280 "\nOn line: " + std::to_string(curLex_.get_line()));
281 }
282 argTypes.push_back(GetVarType(curLex_.get_text()));
283 }
284 AnalyzeExpression();
285 if (curLex_.get_text() != "," && curLex_.get_text() != ")") {
286 throw std::runtime_error(
287 "Expected ',' or ')' after argument \nOn line: " +
288 std::to_string(curLex_.get_line()));
289 }
290 if (curLex_.get_text() == ",") {
291 GetLexem();
292 }
293 }
294 if (argTypes.size() != functionSignatures_[funcName].size()) {
295 throw std::runtime_error(
296 "Invalid number of arguments for function: " + funcName +
297 "\nOn line: " + std::to_string(curLex_.get_line()) +
298 "\nExpected: " + std::to_string(functionSignatures_[funcName].size()) +
299 "\nGot: " + std::to_string(argTypes.size()));
300 }

302 if (argTypes[i] != functionSignatures_[funcName][i]) {
303 throw std::runtime_error(

306 functionSignatures_[funcName][i] + "\nGot: " + argTypes[i]);
307 }
