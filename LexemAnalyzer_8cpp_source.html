134void LexemAnalyzer::HandleIndentation(int currentIndent) {
135 int previousIndent = indentStack_.back();
136 if (currentIndent > previousIndent) {
137 indentStack_.push_back(currentIndent);
138 lexems_.emplace_back(Lexem(LexemType::INDENT, "INDENT",
139 index_ - currentIndent, index_, curLine_));
140 } else {
141 while (currentIndent < previousIndent) {
142 indentStack_.pop_back();
143 lexems_.emplace_back(
144 Lexem(LexemType::DEDENT, "DEDENT", index_, index_, curLine_));
145 lexems_.emplace_back(
146 Lexem(LexemType::NEWLINE, "\\n", index_, index_, curLine_));
147 if (indentStack_.empty()) {
148 throw std::runtime_error(
149 "Indentation error: No matching indentation level.\n On line: " +
150 std::to_string(curLine_));
151 }
152 previousIndent = indentStack_.back();
153 }
154 if (currentIndent != previousIndent) {
155 throw std::runtime_error(
156 "Indentation error: Inconsistent indentation.\n On line: " +
157 std::to_string(curLine_));
158 }
159 }
